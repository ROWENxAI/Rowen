function getUrl(){

$http_type = ((isset($_SERVER['HTTPS']) && $_SERVER['HTTPS'] == 'on') || (isset($_SERVER['HTTP_X_FORWARDED_PROTO']) && $_SERVER['HTTP_X_FORWARDED_PROTO'] == 'https')) ? 'https://' : 'http://';

<?php

namespace app\common\model;

use think\Model;

class User extends Model

{

public function regist($data){

$check = $this->where('name',$data['name'])->find();

if ($check) {

return json_encode(['status'=>'err','msg'=>'用户名已被占用']);

}else{

$res = $this->allowField(true)->save($data);

if($res){

session('user_id',$this->getLastInsID());

session('user_name',$data['name']);

return json_encode(['status'=>'ok','msg'=>'注册成功']);

}else{

return json_encode(['status'=>'err','msg'=>'注册失败']);

}

}

}

public function login($data){

$res = $this->where('name',$data['name'])->where('passwd',$data['passwd'])->find();

if($res){

session('user_id',$res['id']);

session('user_name',$res['name']);

return json_encode(['status'=>'ok','msg'=>'登陆成功']);

}else{

return json_encode(['status'=>'err','msg'=>'用户名或密码错误']);

}

}

}

class PublishAd extends Validate

'award_single'   => 'require|elt:award_total',

'ad_title' => 'require',

];

protected $message  =   [

'title.require'  => '发布标题不能为空',

'sub_title.require'   => '发布副标题不能为空',

'expired_time.require' => '过期时间不能为空',

'award_total.require'  => '奖金总金额不能为空',

'award_single.require'   => '单分金额不能为空',

'award_single。elt'   => '单分金额不能大于奖金总金额',

'ad_title.require' => '内容详情不能为空',

];

// 自定义验证规则

protected function checkExpired($value,$rule,$data=[])

{

if (strtotime($value)>strtotime(date("Y-m-d"))+3600*24*2) {

return true;

}

return '过期时间不能少于三天';

}

}
